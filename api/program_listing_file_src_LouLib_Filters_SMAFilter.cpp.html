#include "SMAFilter.hpp"

namespace LouLib {
    namespace Filters {

        SMAFilter::SMAFilter(int readingCount) : readingCount(readingCount){
            sum = 0;
        }

        void SMAFilter::addReading(double reading) {
            sum += reading;
            data.push(reading);
            if(data.size() > readingCount){
                sum -= data.front();
                data.pop();
            }
        }

        double SMAFilter::getOutput() {
            return sum / data.size();
        }
    } // LouLib
} // Filters
